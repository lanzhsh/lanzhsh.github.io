递归的三种常见应用

By LanZhiSheng

发表于 2016-04-29

概念

函数直接或者间接调用自己

特点

存在一个零界条件,循环和递归一般可以互换

应用

A.求阶乘

function factorial(n) {
    if(n === 1) {
        return 1;
            }else {
        return factorial(n - 1) * n;
    }
}

B.求斐波拉契数列

function fib(n) {
    if (n === 0 || n === 1) {
        return 1;
            } else {
    return fib(n - 2) + fib(n - 1);
    }
}

C.优化版斐波拉契数列

var total = 0;
  var fibCache = [];
    function fib(n) {
      total++;
      return fibCache[n] || (fibCache[n] = 
(n === 0 || n === 1)? 1: fib(n - 2) + fib(n - 1));
    }
console.log(fib(100), total);

D.统计页面中所有的元素

1.思路:
从html根节点开始获取子元素，
把子元素添加到数组中，
然后继续以递归的方式获取子元素的子元素

2.代码:
//递归获取页面上所有的元素
var nodes = [ document.documentElement ];
        (function getNodes(node) {
            var child = node.children;
        for(var i = 0, len = child.length; i < len; 
    i++) {
                nodes.push(child[i]);
                getNodes(child[i]);
            }
})(document.documentElement);

//统计元素的种类
var obj = { total: 0 };
    nodes.forEach(function(node, i) {
        if(!obj[node.nodeName]) {
            obj[node.nodeName] = true;
                obj.total++;
    }
});
